<script>
function getColor(number) {
    if (number <= 10) return "#fbc531";
    if (number <= 20) return "#00a8ff";
    if (number <= 30) return "#4cd137";
    if (number <= 40) return "#8c7ae6";
    return "#e84118";
}

function generateLotto() {
    let numbers = [];

    while (numbers.length < 7) {
        let num = Math.floor(Math.random() * 45) + 1;
        if (!numbers.includes(num)) {
            numbers.push(num);
        }
    }

    let mainNumbers = numbers.slice(0, 6).sort((a, b) => a - b);
    let bonusNumber = numbers[6];

    const container = document.getElementById("lottoNumbers");
    container.innerHTML = "";

    mainNumbers.forEach(num => {
        const span = document.createElement("span");
        span.className = "number";
        span.style.backgroundColor = getColor(num);
        span.textContent = num;
        container.appendChild(span);
    });

    const plus = document.createElement("span");
    plus.textContent = " + ";
    plus.style.fontSize = "24px";
    container.appendChild(plus);

    const bonus = document.createElement("span");
    bonus.className = "number bonus";
    bonus.textContent = bonusNumber;
    container.appendChild(bonus);
}
</script>
